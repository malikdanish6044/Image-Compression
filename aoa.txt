(function() {

  let W, H, particles = [], gridP = [];
  function resize() { W = pc.width = innerWidth; H = pc.height = innerHeight; }
  window.addEventListener('resize', resize); resize();
  const COLORS = ['108,99,255','67,233,123','255,101,132'];
  for (let i=0;i<45;i++) particles.push({ x:Math.random()*W, y:Math.random()*H, r:Math.random()*2+0.5, vx:(Math.random()-0.5)*0.3, vy:(Math.random()-0.5)*0.3, color:COLORS[i%3], alpha:Math.random()*0.5+0.1, pulse:Math.random()*Math.PI*2 });

    ctx.clearRect(0,0,W,H);
    gridP.forEach(g => { g.y-=g.speed; if(g.y+g.size<0){g.y=H+g.size;g.x=Math.random()*W;} ctx.strokeStyle=`rgba(108,99,255,${g.alpha})`; ctx.lineWidth=0.5; ctx.strokeRect(g.x,g.y,g.size,g.size); ctx.strokeRect(g.x,g.y,g.size/2,g.size/2); ctx.strokeRect(g.x+g.size/2,g.y,g.size/2,g.size/2); });

  }

function calcQuality(ratio) {
  if(ratio<20) return {pct:92,label:'Excellent',color:'#43e97b'};
  if(ratio<40) return {pct:75,label:'Good',color:'#43e97b'};

  return {pct:35,label:'Heavy',color:'#ff6584'};
}

// ── QUADTREE ──
function getStats(data, x, y, w, h, W) {
  let rS=0,gS=0,bS=0,n=0;
  for(let j=y;j<y+h;j++) for(let i=x;i<x+w;i++){const idx=(j*W+i)*4;rS+=data[idx];gS+=data[idx+1];bS+=data[idx+2];n++;}
  const rA=rS/n,gA=gS/n,bA=bS/n;
  let v=0;
  for(let j=y;j<y+h;j++) for(let i=x;i<x+w;i++){const idx=(j*W+i)*4;v+=(data[idx]-rA)**2+(data[idx+1]-gA)**2+(data[idx+2]-bA)**2;}
  return {v:v/n,r:Math.round(rA),g:Math.round(gA),b:Math.round(bA)};
}

function runQuadtree(data, ctx, x, y, w, h, depth, threshold, maxDepth) {
  if(w<1||h<1) return;
  nodeCount++;
  if(depth>maxDepthUsed) maxDepthUsed=depth;
  const W=origCanvas.width;
  const {v,r,g,b}=getStats(data,x,y,w,h,W);
  if(v<threshold*threshold*3||depth>=maxDepth||w<=1||h<=1){
    ctx.fillStyle=`rgb(${r},${g},${b})`;
    ctx.fillRect(x,y,w,h);
    if(w>6&&h>6){ctx.strokeStyle='rgba(0,0,0,0.08)';ctx.lineWidth=0.5;ctx.strokeRect(x+0.5,y+0.5,w-1,h-1);}
    return;
  }
  const hw=Math.floor(w/2),hh=Math.floor(h/2);
  runQuadtree(data,ctx,x,   y,   hw,   hh,   depth+1,threshold,maxDepth);
  runQuadtree(data,ctx,x+hw,y,   w-hw, hh,   depth+1,threshold,maxDepth);
  runQuadtree(data,ctx,x,   y+hh,hw,   h-hh, depth+1,threshold,maxDepth);
  runQuadtree(data,ctx,x+hw,y+hh,w-hw, h-hh, depth+1,threshold,maxDepth);
}

// ── VISUALIZATION MODE ──
function runQuadtreeViz(data, ctx, x, y, w, h, depth, threshold, maxDepth) {
  if(w<1||h<1) return;
  const W=origCanvas.width;
  const {v,r,g,b}=getStats(data,x,y,w,h,W);
  if(v<threshold*threshold*3||depth>=maxDepth||w<=1||h<=1){
    ctx.fillStyle=`rgb(${r},${g},${b})`;
    ctx.fillRect(x,y,w,h);
    // Draw grid lines with color based on depth
    const hue=(depth/maxDepth)*240;
    ctx.strokeStyle=`hsla(${hue},80%,65%,0.7)`;
    ctx.lineWidth=depth===0?2:1;
    ctx.strokeRect(x+0.5,y+0.5,w-1,h-1);
    return;
  }
  const hw=Math.floor(w/2),hh=Math.floor(h/2);
  runQuadtreeViz(data,ctx,x,   y,   hw,   hh,   depth+1,threshold,maxDepth);
  runQuadtreeViz(data,ctx,x+hw,y,   w-hw, hh,   depth+1,threshold,maxDepth);
  runQuadtreeViz(data,ctx,x,   y+hh,hw,   h-hh, depth+1,threshold,maxDepth);
  runQuadtreeViz(data,ctx,x+hw,y+hh,w-hw, h-hh, depth+1,threshold,maxDepth);
}

function updateVizView() {
  if(!origImageData) return;
  vizCanvas.width=origCanvas.width; vizCanvas.height=origCanvas.height;
  const ctx=vizCanvas.getContext('2d');
  ctx.fillStyle='#000'; ctx.fillRect(0,0,vizCanvas.width,vizCanvas.height);
  runQuadtreeViz(origImageData.data, ctx, 0, 0, origCanvas.width, origCanvas.height, 0, parseInt(threshSlider.value), parseInt(depthSlider.value));
  showImageOnCard('viz-body', vizCanvas);
  document.getElementById('dl-viz').disabled=false;
}

// ── SPLIT VIEW (single-canvas draw approach) ──
let splitCanvas = null; // will be set on first use

function updateSplitView() {
  if(!origImageData || compCanvas.width===0) return;

  const wrap = document.getElementById('split-wrap');
  const dispCanvas = document.getElementById('split-display-canvas');
  splitCanvas = dispCanvas;

  const maxW = wrap.clientWidth || 700;
  const maxH = 520;
  const scale = Math.min(maxW / origCanvas.width, maxH / origCanvas.height, 1);
  const dispW = Math.round(origCanvas.width * scale);
  const dispH = Math.round(origCanvas.height * scale);

  // Set display canvas size = pixel dimensions for sharp rendering
  dispCanvas.width = dispW;
  dispCanvas.height = dispH;
  dispCanvas.style.display = 'block';
  dispCanvas.style.width = dispW + 'px';
  dispCanvas.style.height = dispH + 'px';

  wrap.style.height = dispH + 'px';

  // Draw both sides
  drawSplit(dispW, dispH);

  // Show UI elements
  document.getElementById('split-placeholder').style.display = 'none';
  document.getElementById('split-divider').style.display = 'block';
  document.getElementById('split-lbl-orig').style.display = 'block';
  document.getElementById('split-lbl-comp').style.display = 'block';

  // Position divider
  updateSplitDivider(dispW, dispH);
  document.getElementById('dl-split-png').disabled = false;
  document.getElementById('dl-split-jpg').disabled = false;
}

function drawSplit(dispW, dispH) {
  if(!splitCanvas) return;
  const ctx = splitCanvas.getContext('2d');
  const splitPx = Math.round(dispW * splitRatio);

  // Draw full original image on left side
  ctx.drawImage(origCanvas, 0, 0, origCanvas.width, origCanvas.height, 0, 0, dispW, dispH);

  // Draw compressed on right side (using clip)
  ctx.save();
  ctx.beginPath();
  ctx.rect(splitPx, 0, dispW - splitPx, dispH);
  ctx.clip();
  ctx.drawImage(compCanvas, 0, 0, compCanvas.width, compCanvas.height, 0, 0, dispW, dispH);
  ctx.restore();

  // Draw divider line ON the canvas itself
  ctx.save();
  ctx.strokeStyle = 'rgba(255,255,255,0.9)';
  ctx.lineWidth = 2;
  ctx.shadowColor = 'rgba(0,0,0,0.5)';
  ctx.shadowBlur = 6;
  ctx.beginPath();
  ctx.moveTo(splitPx, 0);
  ctx.lineTo(splitPx, dispH);
  ctx.stroke();
  ctx.restore();
}

function updateSplitDivider(dispW, dispH) {
  const wrap = document.getElementById('split-wrap');
  const divider = document.getElementById('split-divider');
  const wrapW = wrap.clientWidth;
  // Center the image in the wrap
  const offsetX = (wrapW - dispW) / 2;
  const splitPx = offsetX + dispW * splitRatio;
  divider.style.left = splitPx + 'px';
}

// Mouse / touch events on the WRAP (not just divider)
const splitWrap = document.getElementById('split-wrap');

function getSplitRatioFromEvent(clientX) {
  const wrap = document.getElementById('split-wrap');
  const rect = wrap.getBoundingClientRect();
  if(!splitCanvas) return splitRatio;
  const dispW = splitCanvas.width;
  const offsetX = (rect.width - dispW) / 2;
  const relX = clientX - rect.left - offsetX;
  return Math.max(0.02, Math.min(0.98, relX / dispW));
}

splitWrap.addEventListener('mousedown', e => { isDraggingSplit = true; e.preventDefault(); });
document.addEventListener('mousemove', e => {
  if(!isDraggingSplit) return;
  splitRatio = getSplitRatioFromEvent(e.clientX);
  if(splitCanvas) {
    drawSplit(splitCanvas.width, splitCanvas.height);
    updateSplitDivider(splitCanvas.width, splitCanvas.height);
  }
});
document.addEventListener('mouseup', () => isDraggingSplit = false);

splitWrap.addEventListener('touchstart', e => { isDraggingSplit = true; e.preventDefault(); }, {passive:false});
document.addEventListener('touchmove', e => {
  if(!isDraggingSplit) return;
  splitRatio = getSplitRatioFromEvent(e.touches[0].clientX);
  if(splitCanvas) {
    drawSplit(splitCanvas.width, splitCanvas.height);
    updateSplitDivider(splitCanvas.width, splitCanvas.height);
  }
}, {passive:true});
document.addEventListener('touchend', () => isDraggingSplit = false);

// ── HISTORY ──
function saveHistory(ratio, origKB, compKB, threshold, depth) {
  const thumb=document.createElement('canvas');
  thumb.width=44; thumb.height=44;
  const tc=thumb.getContext('2d');
  tc.drawImage(compCanvas,0,0,44,44);

  compressionHistory.unshift({thumb:thumb.toDataURL(),ratio,origKB,compKB,threshold,depth,time:new Date().toLocaleTimeString()});
  if(compressionHistory.length>4) compressionHistory.pop();
  renderHistory();
}

function renderHistory() {
  const list=document.getElementById('history-list');
  if(!compressionHistory.length){ list.innerHTML='<div class="history-empty">Abhi koi history nahi</div>'; return; }
  list.innerHTML=compressionHistory.map((h,i)=>`
    <div class="history-item" onclick="loadHistory(${i})">
      <img class="history-thumb" src="${h.thumb}" alt="thumb">
      <div class="history-meta">
        <div class="history-name">${h.time}</div>
        <div class="history-stats">T:${h.threshold} D:${h.depth} · ${formatSize(h.origKB)}→${formatSize(h.compKB)}</div>
      </div>
      <div class="history-ratio">${h.ratio}%</div>
    </div>
  `).join('');
}

function loadHistory(i) {
  const h=compressionHistory[i];
  threshSlider.value=h.threshold;
  depthSlider.value=h.depth;
  document.getElementById('thresh-val').textContent=h.threshold;
  document.getElementById('depth-val').textContent=h.depth;
  showToast(`📂 History loaded — Threshold:${h.threshold}, Depth:${h.depth}`);
  runCompression(false);
}
